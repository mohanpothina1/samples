private void updateUIWithData(int startIndex, int moduleIndex, HardwareData data) {
    JTextField vswrField = vswrList.get(startIndex + moduleIndex);
    
    if (!data.paOn) {
        vswrField.setText("--");
        vswrField.setBackground(Color.WHITE);
        vswrField.setForeground(Color.BLACK);
        vswrField.setToolTipText(null);
        therList.get(startIndex + moduleIndex).setText("--");
        fwdList.get(startIndex + moduleIndex).setText("--");
        rsvList.get(startIndex + moduleIndex).setText("--");
        return;
    }
    
    double vswr = data.vswr;
    vswrField.setText(String.valueOf(vswr));
    
    if (vswr >= 3.5) {
        vswrField.setBackground(Color.RED);
        vswrField.setForeground(Color.WHITE);
        vswrField.setToolTipText("VSWR CRITICAL");

        vswrField.setForeground(Color.BLACK);
        vswrField.setToolTipText("Warning");
    } else {
        vswrField.setBackground(Color.GREEN);
        vswrField.setForeground(Color.BLACK);

    therList.get(startIndex + moduleIndex).setText(String.valueOf(data.temp));
    fwdList.get(startIndex + moduleIndex).setText(String.valueOf(data.fwd));
    rsvList.get(startIndex + moduleIndex).setText(String.valueOf(data.rfl));

private boolean isSpecialBand(String name) {
    return name.equals("136-174MHz") ||
           name.equals("430-470MHz") ||
           name.equals("100-500MHz") ||
           name.equals("2400-2500MHz");
}

private void clearModuleData(int startIndex, int moduleIndex) {
    int idx = startIndex + moduleIndex;
    if (idx >= vswrList.size()) return;
    
    vswrList.get(idx).setText("--");
    vswrList.get(idx).setBackground(Color.WHITE);
    vswrList.get(idx).setForeground(Color.BLACK);
    therList.get(idx).setText("--");
    fwdList.get(idx).setText("--");
    rsvList.get(idx).setText("--");
}

private void updateModuleData(int startIndex, int moduleIndex, HardwareData data) {
    int idx = startIndex + moduleIndex;
    if (idx >= vswrList.size()) return;
    
    if (!data.paOn) {
        clearModuleData(startIndex, moduleIndex);
        return;
    }
    
    JTextField vswrField = vswrList.get(idx);
    double vswr = data.vswr;
    vswrField.setText(String.valueOf(vswr));
    
    if (vswr >= 3.5) {
        vswrField.setBackground(Color.RED);
        vswrField.setForeground(Color.WHITE);
        vswrField.setToolTipText("VSWR CRITICAL");
    } else if (vswr >= 2.5) {
        vswrField.setBackground(Color.YELLOW);
        vswrField.setForeground(Color.BLACK);
        vswrField.setToolTipText("Warning");
    } else {
        vswrField.setBackground(Color.GREEN);
        vswrField.setForeground(Color.BLACK);
        vswrField.setToolTipText(null);
    }
    
    therList.get(idx).setText(String.valueOf(data.temp));
    fwdList.get(idx).setText(String.valueOf(data.fwd));
    rsvList.get(idx).setText(String.valueOf(data.rfl));
}
