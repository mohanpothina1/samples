// Method to update health status UI
private void updateHealthStatusUI(String cardId, HealthStatusData data) {
    // Find or create health status panel for this card
    String panelKey = "health_" + cardId;
    
    // Update the labels in the health status panel
    // These labels should be stored in a map when created
    JLabel paLabel = healthStatusLabels.get(cardId + "_pa");
    JLabel vswrLabel = healthStatusLabels.get(cardId + "_vswr");
    JLabel tempLabel = healthStatusLabels.get(cardId + "_temp");
    JLabel fwdLabel = healthStatusLabels.get(cardId + "_fwd");
    JLabel refLabel = healthStatusLabels.get(cardId + "_ref");
    
    if (paLabel != null) {
        paLabel.setText(data.paOnOff ? "ON" : "OFF");
        paLabel.setForeground(data.paOnOff ? new Color(0, 128, 0) : Color.RED);
        paLabel.setFont(new Font("Segoe UI", Font.BOLD, 12));
    }
    
    if (vswrLabel != null) {
        vswrLabel.setText(data.vswrAlarm == 0 ? "OK" : "ALERT");
        vswrLabel.setForeground(data.vswrAlarm == 0 ? new Color(0, 128, 0) : Color.RED);
        vswrLabel.setFont(new Font("Segoe UI", Font.BOLD, 12));
        if (data.vswrAlarm != 0) {
            vswrLabel.setToolTipText("VSWR Alarm! Check antenna connection");
        }
    }
    
    if (tempLabel != null) {
        tempLabel.setText(data.tempAlarm == 0 ? "OK" : "ALERT");
        tempLabel.setForeground(data.tempAlarm == 0 ? new Color(0, 128, 0) : Color.RED);
        tempLabel.setFont(new Font("Segoe UI", Font.BOLD, 12));
        if (data.tempAlarm != 0) {
            tempLabel.setToolTipText("Temperature Alarm! Check cooling system");
        }
    }
    
    if (fwdLabel != null) {
        fwdLabel.setText(String.format("%.1f W", data.fwdPower));
        fwdLabel.setForeground(data.fwdPower > 0 ? new Color(0, 128, 0) : Color.GRAY);
        fwdLabel.setFont(new Font("Segoe UI", Font.BOLD, 12));
    }
    
    if (refLabel != null) {
        refLabel.setText(String.format("%.1f W", data.refPower));
        refLabel.setForeground(data.refPower < 5 ? new Color(0, 128, 0) : Color.RED);
        refLabel.setFont(new Font("Segoe UI", Font.BOLD, 12));
        if (data.refPower >= 5) {
            refLabel.setToolTipText("High reflected power! Check antenna");
        }
    }
}

// Add this method to create health status panel for each card in the UI
private JPanel createHealthStatusPanel(String cardId) {
    JPanel healthPanel = new JPanel();
    healthPanel.setBorder(new TitledBorder(new LineBorder(new Color(0, 0, 0)), 
        "Health - " + cardId, 
        TitledBorder.LEFT, TitledBorder.TOP, 
        new Font("Segoe UI", Font.BOLD, 11), new Color(0, 0, 0)));
    healthPanel.setLayout(new GridLayout(5, 2, 5, 3));
    healthPanel.setPreferredSize(new Dimension(180, 120));
    healthPanel.setBackground(Color.WHITE);
    
    // Create labels for health status
    String[] labelNames = {"PA:", "VSWR:", "TEMP:", "FWD:", "REF:"};
    JLabel[] valueLabels = new JLabel[5];
    
    for (int i = 0; i < labelNames.length; i++) {
        JLabel nameLabel = new JLabel(labelNames[i]);
        nameLabel.setFont(new Font("Segoe UI", Font.PLAIN, 11));
        healthPanel.add(nameLabel);
        
        valueLabels[i] = new JLabel("--");
        valueLabels[i].setFont(new Font("Segoe UI", Font.BOLD, 11));
        valueLabels[i].setHorizontalAlignment(SwingConstants.RIGHT);
        healthPanel.add(valueLabels[i]);
    }
    
    // Store the value labels for updates
    healthStatusLabels.put(cardId + "_pa", valueLabels[0]);
    healthStatusLabels.put(cardId + "_vswr", valueLabels[1]);
    healthStatusLabels.put(cardId + "_temp", valueLabels[2]);
    healthStatusLabels.put(cardId + "_fwd", valueLabels[3]);
    healthStatusLabels.put(cardId + "_ref", valueLabels[4]);
    
    return healthPanel;
}

private void setupHealthStatusPanels(JPanel eastPanel, List<ControlCard> allCards) {
    JPanel healthContainer = new JPanel();
    healthContainer.setBorder(new TitledBorder(new LineBorder(new Color(0, 0, 0)), 
        "Health Status", TitledBorder.CENTER, TitledBorder.TOP, 
        new Font("Segoe UI", Font.BOLD, 14), new Color(0, 128, 0)));
    healthContainer.setLayout(new GridLayout(0, 1, 5, 5));
    
    // Create health status panel for each control card
    for (ControlCard card : allCards) {
        JPanel healthPanel = createHealthStatusPanel(card.id);
        healthContainer.add(healthPanel);
    }
    
    // Add health container to the east panel (at the bottom or in a specific position)
    eastPanel.add(healthContainer, BorderLayout.SOUTH);
}
