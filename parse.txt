private HardwareData parseFrame(byte[] frame) {

    HardwareData data = new HardwareData();

    if (frame.length < 14) {
        System.out.println("Packet too short for parsing.");
        return data;
    }

    int payLoad = 6;  // Start after header (7E 16 04 13 00 1B)

    // Skip Status (offset 1) and Fault Status (offset 2)
    payLoad++;  // Status = frame[6]
    payLoad++;  // Fault Status = frame[7]

    // Skip Voltage (offset 3-4)
    payLoad += 2;  // Voltage = frame[8-9]

    // Skip Current (offset 5-6)
    payLoad += 2;  // Current = frame[10-11]

    // === TEMPERATURE (offset 7) ===
    data.temp = frame[payLoad++] & 0xFF;  // 37°C

    // === FORWARD POWER (offset 8-9) ===
    int fwdRaw = ((frame[payLoad++] & 0xFF) << 8) | (frame[payLoad++] & 0xFF);
    
    // === REFLECTED POWER (offset 10-11) ===
    int rflRaw = ((frame[payLoad++] & 0xFF) << 8) | (frame[payLoad++] & 0xFF);

    // Store raw values
    data.fwd = fwdRaw;
    data.rfl = rflRaw;

    // === CONVERT TO WATTS ===
    // Assuming raw values are in 0.1 dBm steps (typical for these modules)
    // Formula: Power (W) = 10^((dBm - 30) / 10)
    // If raw is in 0.1 dBm: dBm = raw / 10
    
    double fwd_dBm = fwdRaw / 10.0;  // Convert from 0.1 dBm to dBm
    double rfl_dBm = rflRaw / 10.0;
    
    data.fwdWatts = Math.pow(10, (fwd_dBm - 30) / 10);  // dBm to Watts
    data.rflWatts = Math.pow(10, (rfl_dBm - 30) / 10);
    
    // Handle special cases (0 dBm = 0.001W = 1mW)
    if (fwdRaw == 0) {
        data.fwdWatts = 0.0;
    }
    if (rflRaw == 0) {
        data.rflWatts = 0.0;
    }

    // === CALCULATE VSWR ===
    if (fwdRaw > 0 && data.fwdWatts > 0) {
        // Calculate VSWR from power ratio
        double powerRatio = data.rflWatts / data.fwdWatts;
        
        // Clamp to valid range
        if (powerRatio < 1.0) {
            double gamma = Math.sqrt(powerRatio);
            data.vswr = (1 + gamma) / (1 - gamma);
            
            // Clamp VSWR to reasonable range
            if (data.vswr > 99.9) {
                data.vswr = 99.9;
            } else if (data.vswr < 1.0) {
                data.vswr = 1.0;
            }
        } else {
            data.vswr = 99.9;  // Infinite VSWR
        }
    } else {
        data.vswr = 1.0;  // No forward power = no reflection
    }

    data.paOn = (fwdRaw > 0);

    return data;
}

private HardwareData parseFrame(byte[] frame) {

    HardwareData data = new HardwareData();

    if (frame == null || frame.length < 35) {
        System.out.println("Packet too short.");
        return data;
    }

    // Payload starts after:
    // SOF(1) + Func(1) + Addr(1) + Cmd(1) + Ack(1) + Length(1)
    int index = 6;

    // Heartbeat
    int heartbeat = frame[index++] & 0xFF;

    // Status
    int status = frame[index++] & 0xFF;

    // Fault Status
    int fault = frame[index++] & 0xFF;

    // Voltage (2 bytes)
    int voltage = ((frame[index++] & 0xFF) << 8)
                | (frame[index++] & 0xFF);

    // Current (2 bytes)
    int current = ((frame[index++] & 0xFF) << 8)
                | (frame[index++] & 0xFF);

    // Temperature (1 byte)
    data.temp = frame[index++] & 0xFF;

    // Forward Power (2 bytes)
    data.fwd = ((frame[index++] & 0xFF) << 8)
             | (frame[index++] & 0xFF);

    // Reverse Voltage (2 bytes)
    int reverseVoltage = ((frame[index++] & 0xFF) << 8)
                       | (frame[index++] & 0xFF);

    // ALC (2 bytes)
    int alc = ((frame[index++] & 0xFF) << 8)
            | (frame[index++] & 0xFF);

    // Forward Sampling Voltage (2 bytes)
    int forwardSample = ((frame[index++] & 0xFF) << 8)
                      | (frame[index++] & 0xFF);

    // Reverse Sampling Voltage (2 bytes)
    int reverseSample = ((frame[index++] & 0xFF) << 8)
                      | (frame[index++] & 0xFF);

    // Store values
    data.rfl = reverseVoltage;   // This is Reverse Voltage, NOT reflected RF power
    data.paOn = data.fwd > 0;

    // Debug
    System.out.println("Heartbeat      : " + heartbeat);
    System.out.println("Status         : " + status);
    System.out.println("Fault          : " + fault);
    System.out.println("Voltage        : " + voltage);
    System.out.println("Current        : " + current);
    System.out.println("Temperature    : " + data.temp + " °C");
    System.out.println("Forward Power  : " + data.fwd);
    System.out.println("Reverse Voltage: " + reverseVoltage);
    System.out.println("ALC            : " + alc);
    System.out.println("Forward Sample : " + forwardSample);
    System.out.println("Reverse Sample : " + reverseSample);

    return data;
}
