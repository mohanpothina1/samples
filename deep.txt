// Add this new class for dual port handling
class DualPortHandler implements Runnable {
    private final String ip;
    private final int controlPort; // port 7 for control
    private static final int STATUS_PORT = 20108; // port for status
    private final byte[] statusCommand;
    private final Consumer<Boolean> statusCallback;
    private final Consumer<HardwareData> dataCallback;
    private volatile boolean running = true;
    private volatile boolean connected = false;
    private Socket controlSocket;
    private Socket statusSocket;
    private OutputStream controlOut;
    private OutputStream statusOut;
    private final ArrayBlockingQueue<byte[]> sendQueue = new ArrayBlockingQueue<>(100);
    private volatile HardwareData lastData;
    private Thread senderThread;
    private ScheduledExecutorService scheduler;
    private boolean statusEnabled = false;

    public DualPortHandler(String ip, int port, byte[] statusCommand, 
                          Consumer<Boolean> statusCallback, 
                          Consumer<HardwareData> dataCallback) {
        this.ip = ip;
        this.controlPort = port;
        this.statusCommand = statusCommand;
        this.statusCallback = statusCallback;
        this.dataCallback = dataCallback;
    }

    @Override
    public void run() {
        while (running) {
            try {
                // Step 1: Connect to status port 20108
                statusSocket = new Socket();
                statusSocket.connect(new InetSocketAddress(ip, STATUS_PORT), 3000);
                statusOut = statusSocket.getOutputStream();
                
                // Step 2: Send command to enable PA status
                byte[] enableStatusCmd = "CO.PA.S,1".getBytes(); // Enable PA status
                statusOut.write(enableStatusCmd);
                statusOut.flush();
                Thread.sleep(500); // Wait for response
                
                // Step 3: Get default values
                statusOut.write(statusCommand);
                statusOut.flush();
                
                // Start status receiver thread
                Thread statusReceiver = new Thread(() -> {
                    try {
                        InputStream in = statusSocket.getInputStream();
                        while (running && !statusSocket.isClosed()) {
                            byte[] frame = readFrame(in);
                            if (frame != null && verifyXor(frame)) {
                                HardwareData data = parseFrame(frame);
                                this.lastData = data;
                                if (dataCallback != null) {
                                    dataCallback.accept(data);
                                }
                            }
                        }
                    } catch (Exception e) {
                        System.out.println("Status receiver stopped");
                    }
                });
                statusReceiver.start();
                
                // Step 4: Connect to control port (7)
                controlSocket = new Socket();
                controlSocket.connect(new InetSocketAddress(ip, controlPort), 3000);
                controlOut = controlSocket.getOutputStream();
                
                connected = true;
                if (statusCallback != null) {
                    statusCallback.accept(true);
                }
                
                // Start sender for control commands
                startSender();
                
                // Start auto polling for status
                startAutoPolling();
                
                // Wait for disconnection
                while (running && controlSocket != null && !controlSocket.isClosed()) {
                    Thread.sleep(200);
                }
                
            } catch (Exception e) {
                System.out.println("Dual port connection failed for: " + ip);
                connected = false;
                if (statusCallback != null) {
                    statusCallback.accept(false);
                }
                closeSockets();
                try {
                    Thread.sleep(3000);
                } catch (InterruptedException ignored) {}
            }
        }
    }

    private void startSender() {
        senderThread = new Thread(() -> {
            try {
                while (running && controlSocket != null && !controlSocket.isClosed()) {
                    byte[] cmd = sendQueue.take();
                    if (controlOut != null) {
                        controlOut.write(cmd);
                        controlOut.flush();
                        System.out.println("Sent to control port: " + bytesToHex(cmd));
                    }
                }
            } catch (Exception e) {
                System.out.println("Sender stopped.");
            }
        });
        senderThread.start();
    }

    private void startAutoPolling() {
        if (statusCommand == null) return;
        scheduler = Executors.newSingleThreadScheduledExecutor();
        scheduler.scheduleAtFixedRate(() -> {
            if (connected && statusOut != null) {
                try {
                    statusOut.write(statusCommand);
                    statusOut.flush();
                } catch (IOException e) {
                    System.out.println("Status polling failed");
                }
            }
        }, 0, 1, TimeUnit.SECONDS);
    }

    private byte[] readFrame(InputStream in) throws IOException {
        ByteArrayOutputStream buffer = new ByteArrayOutputStream();
        int b;
        
        // Wait for frame start (0x5A)
        while ((b = in.read()) != -1) {
            if (b == 0x5A) {
                buffer.write(b);
                break;
            }
        }
        if (b == -1) return null;
        
        // Read until frame end (0xA5)
        while ((b = in.read()) != -1) {
            buffer.write(b);
            if (b == 0xA5) {
                break;
            }
        }
        return buffer.toByteArray();
    }

    private boolean verifyXor(byte[] frame) {
        if (frame.length < 3) return false;
        int xor = 0;
        for (int i = 0; i < frame.length - 2; i++) {
            xor ^= frame[i];
        }
        return xor == (frame[frame.length - 2] & 0xFF);
    }

    private HardwareData parseFrame(byte[] frame) {
        HardwareData data = new HardwareData();
        data.paOn = false;
        int i = 0;
        
        while (i < frame.length) {
            int id = frame[i] & 0xFF;
            
            if (id == 0xF1) {
                data.fwd = frame[i + 2] & 0xFF;
                if (data.fwd > 0) data.paOn = true;
                i += 3;
            } else if (id == 0xF2) {
                data.rfl = frame[i + 2] & 0xFF;
                i += 3;
            } else if (id == 0x20) {
                data.temp = ((frame[i + 2] & 0xFF) << 8) | (frame[i + 3] & 0xFF);
                i += 4;
            } else if (id == 0x21) {
                int raw = ((frame[i + 2] & 0xFF) << 8) | (frame[i + 3] & 0xFF);
                data.vswr = raw / 10.0;
                i += 4;
            } else {
                i++;
            }
        }
        return data;
    }

    public void enqueueCommand(byte[] command) {
        if (command != null && connected) {
            sendQueue.offer(command);
        }
    }

    public boolean isConnected() {
        return connected;
    }

    public HardwareData getLastData() {
        return lastData;
    }

    public void start() {
        new Thread(this).start();
    }

    public void shutdown() {
        running = false;
        closeSockets();
        if (scheduler != null) {
            scheduler.shutdownNow();
        }
    }

    private void closeSockets() {
        connected = false;
        try {
            if (controlSocket != null) controlSocket.close();
        } catch (Exception ignored) {}
        try {
            if (statusSocket != null) statusSocket.close();
        } catch (Exception ignored) {}
    }

    private String bytesToHex(byte[] bytes) {
        StringBuilder sb = new StringBuilder();
        for (byte b : bytes) {
            sb.append(String.format("%02X ", b));
        }
        return sb.toString();
    }
}

	private void powerOffAllPAs(Config config) {
		for (PowerSupply psu : config.power_supplies) {
			PowerSupplySocketHandler psuHandler = psuHandlers.get(psu.id);
			if (psuHandler != null && psuHandler.isConnected()) {

				if (psu.id.equals("psu_02")) {
					psuHandler.enqueueCommand("CO.PS.S,0,0".getBytes());
				} else {

					for (ControlCard card : psu.control_cards) {
						if (card.off_command != null && !card.off_command.isEmpty()) {
							splitCmd(card.off_command, psuHandler);
							psuHandler.enqueueCommand(hexStringToBytes(card.off_command));
						} else {
							for (Module m : card.modules) {
								psuHandler.enqueueCommand(hexStringToBytes(m.power_off));
							}
						}
					}
				}
			}
		}
		for (ControlCard card : config.standalone_control_cards) {
			ControlCardSocketHandler handler = cardHandlers.get(card.id);
			if (handler != null && handler.isConnected()) {
				for (Module m : card.modules) {
					handler.enqueueCommand(hexStringToBytes(m.power_off));
				}
			}
		}
	}

	private byte[] hexStringToBytes(String hex) {
		if (hex == null || hex.trim().isEmpty()) {
			return new byte[0];
		}

		String[] parts = hex.trim().split(",");
		List<Byte> byteList = new ArrayList<>();

		for (String part : parts) {
			String value = part.trim();

			if (!value.isEmpty()) { // skip empty values
				byteList.add((byte) Integer.parseInt(value, 16));
			}
		}

		byte[] bytes = new byte[byteList.size()];
		for (int i = 0; i < byteList.size(); i++) {
			bytes[i] = byteList.get(i);
		}

		return bytes;
	}

		for (ControlCardSocketHandler handler : cardHandlers.values()) {
			handler.shutdown();
		}
		for (PowerSupplySocketHandler handler : psuHandlers.values()) {
			handler.shutdown();
		}
		super.dispose();
	}

				doc.insertString(doc.getLength(), msg + "\n", attr);
			} catch (BadLocationException e) {
				// TODO Auto-generated catch block
				e.printStackTrace();
			}
		});
	}

	private void updateTime() {
		SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy hh:mm:ss a");
		label_Time.setText(sdf.format(new Date()));
	}

	private void splitCmd(String command, ControlCardSocketHandler handler) {

		String delimeter = "0d,0a";
		String[] parts = command.split(delimeter);

		for (String part : parts) {
			if (!part.trim().isEmpty()) {
				byte[] cmd = hexStringToBytes(part);
				handler.enqueueCommand(cmd);
			}
		}
	}

	private void updateCardStatus(String cardId, boolean isConnected) {
		SwingUtilities.invokeLater(() -> {

			JLabel lbl = statusLabels.get(cardId);

			if (lbl != null) {
				lbl.setBackground(isConnected ? Color.GREEN : Color.RED);
				lbl.setOpaque(true);
				lbl.revalidate();
				lbl.repaint();
			} else {
				System.out.println("Label not found for: " + cardId);
			}
		});
	}

	private void splitCmd(String command, PowerSupplySocketHandler psuhandler) {
		String delimeter = "0d,0a";
		int start = 0;
		while (true) {
			int index = command.indexOf(delimeter, start);
			if (index == -1)
				break;

			String cmdStr = command.substring(start, index);

			if (!cmdStr.trim().isEmpty()) {
				byte[] cmdBytes = hexStringToBytes(cmdStr);
				psuhandler.enqueueCommand(cmdBytes);
			}
			start = index + delimeter.length() + 1;
		}
	}

import java.io.*;
import java.net.*;
import java.util.concurrent.*;
import java.util.function.Consumer;

public class DualPortHandler implements Runnable {
    private final String ip;
    private final int controlPort;
    private static final int STATUS_PORT = 20108;
    private final byte[] statusEnableCommand;
    private final byte[] statusCommand;
    private final Consumer<Boolean> statusCallback;
    private final Consumer<HardwareData> dataCallback;
    private volatile boolean running = true;
    private volatile boolean connected = false;
    private Socket controlSocket;
    private Socket statusSocket;
    private OutputStream controlOut;
    private OutputStream statusOut;
    private final ArrayBlockingQueue<byte[]> sendQueue = new ArrayBlockingQueue<>(100);
    private volatile HardwareData lastData;
    private Thread senderThread;
    private ScheduledExecutorService scheduler;

    public DualPortHandler(String ip, int controlPort, byte[] statusEnableCommand, 
                          byte[] statusCommand, Consumer<Boolean> statusCallback, 
                          Consumer<HardwareData> dataCallback) {
        this.ip = ip;
        this.controlPort = controlPort;
        this.statusEnableCommand = statusEnableCommand;
        this.statusCommand = statusCommand;
        this.statusCallback = statusCallback;
        this.dataCallback = dataCallback;
    }

    @Override
    public void run() {
        while (running) {
            try {
                // Step 1: Connect to status port 20108
                System.out.println("Connecting to status port " + STATUS_PORT + " for " + ip);
                statusSocket = new Socket();
                statusSocket.connect(new InetSocketAddress(ip, STATUS_PORT), 5000);
                statusOut = statusSocket.getOutputStream();
                
                // Step 2: Send command to enable PA status
                if (statusEnableCommand != null && statusEnableCommand.length > 0) {
                    System.out.println("Enabling PA status...");
                    statusOut.write(statusEnableCommand);
                    statusOut.flush();
                    Thread.sleep(500); // Wait for response
                }
                
                // Step 3: Get default values
                System.out.println("Requesting default values...");
                statusOut.write(statusCommand);
                statusOut.flush();
                
                // Start status receiver thread
                Thread statusReceiver = new Thread(() -> {
                    try {
                        InputStream in = statusSocket.getInputStream();
                        while (running && !statusSocket.isClosed()) {
                            byte[] frame = readFrame(in);
                            if (frame != null && verifyXor(frame)) {
                                HardwareData data = parseFrame(frame);
                                this.lastData = data;
                                if (dataCallback != null) {
                                    dataCallback.accept(data);
                                }
                                System.out.println("Received data: VSWR=" + data.vswr + 
                                                   " TEMP=" + data.temp + 
                                                   " FWD=" + data.fwd + 
                                                   " RFL=" + data.rfl);
                            }
                        }
                    } catch (Exception e) {
                        System.out.println("Status receiver stopped for " + ip);
                    }
                });
                statusReceiver.start();
                
                // Step 4: Connect to control port
                System.out.println("Connecting to control port " + controlPort + " for " + ip);
                controlSocket = new Socket();
                controlSocket.connect(new InetSocketAddress(ip, controlPort), 5000);
                controlOut = controlSocket.getOutputStream();
                
                connected = true;
                if (statusCallback != null) {
                    statusCallback.accept(true);
                }
                
                // Start sender for control commands
                startSender();
                
                // Start auto polling for status
                startAutoPolling();
                
                System.out.println("Dual port connection established for " + ip);
                
                // Wait for disconnection
                while (running && controlSocket != null && !controlSocket.isClosed()) {
                    Thread.sleep(200);
                }
                
            } catch (Exception e) {
                System.out.println("Dual port connection failed for: " + ip + " - " + e.getMessage());
                connected = false;
                if (statusCallback != null) {
                    statusCallback.accept(false);
                }
                closeSockets();
                try {
                    Thread.sleep(3000);
                } catch (InterruptedException ignored) {}
            }
        }
    }

    private void startSender() {
        senderThread = new Thread(() -> {
            try {
                while (running && controlSocket != null && !controlSocket.isClosed()) {
                    byte[] cmd = sendQueue.take();
                    if (controlOut != null) {
                        controlOut.write(cmd);
                        controlOut.flush();
                        System.out.println("Sent to control port: " + bytesToHex(cmd));
                    }
                }
            } catch (Exception e) {
                System.out.println("Sender stopped for " + ip);
            }
        });
        senderThread.start();
    }

    private void startAutoPolling() {
        if (statusCommand == null || statusCommand.length == 0) return;
        scheduler = Executors.newSingleThreadScheduledExecutor();
        scheduler.scheduleAtFixedRate(() -> {
            if (connected && statusOut != null) {
                try {
                    statusOut.write(statusCommand);
                    statusOut.flush();
                } catch (IOException e) {
                    System.out.println("Status polling failed for " + ip);
                }
            }
        }, 0, 1, TimeUnit.SECONDS);
    }

    private byte[] readFrame(InputStream in) throws IOException {
        ByteArrayOutputStream buffer = new ByteArrayOutputStream();
        int b;
        
        // Wait for frame start (0x5A)
        int timeoutCount = 0;
        while ((b = in.read()) != -1) {
            if (b == 0x5A) {
                buffer.write(b);
                break;
            }
            timeoutCount++;
            if (timeoutCount > 1000) return null;
        }
        if (b == -1) return null;
        
        // Read until frame end (0xA5)
        while ((b = in.read()) != -1) {
            buffer.write(b);
            if (b == 0xA5) {
                break;
            }
            if (buffer.size() > 2048) return null;
        }
        return buffer.toByteArray();
    }

    private boolean verifyXor(byte[] frame) {
        if (frame.length < 3) return false;
        int xor = 0;
        for (int i = 0; i < frame.length - 2; i++) {
            xor ^= frame[i];
        }
        return xor == (frame[frame.length - 2] & 0xFF);
    }

    private HardwareData parseFrame(byte[] frame) {
        HardwareData data = new HardwareData();
        data.paOn = false;
        int i = 0;
        
        while (i < frame.length) {
            int id = frame[i] & 0xFF;
            
            if (id == 0xF1) {
                data.fwd = frame[i + 2] & 0xFF;
                if (data.fwd > 0) data.paOn = true;
                i += 3;
            } else if (id == 0xF2) {
                data.rfl = frame[i + 2] & 0xFF;
                i += 3;
            } else if (id == 0x20) {
                data.temp = ((frame[i + 2] & 0xFF) << 8) | (frame[i + 3] & 0xFF);
                i += 4;
            } else if (id == 0x21) {
                int raw = ((frame[i + 2] & 0xFF) << 8) | (frame[i + 3] & 0xFF);
                data.vswr = raw / 10.0;
                i += 4;
            } else {
                i++;
            }
        }
        return data;
    }

    public void enqueueCommand(byte[] command) {
        if (command != null && connected) {
            try {
                sendQueue.offer(command, 100, TimeUnit.MILLISECONDS);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        }
    }

    public boolean isConnected() {
        return connected;
    }

    public HardwareData getLastData() {
        return lastData;
    }

    public void start() {
        new Thread(this).start();
    }

    public void shutdown() {
        running = false;
        closeSockets();
        if (scheduler != null) {
            scheduler.shutdownNow();
        }
        if (senderThread != null) {
            senderThread.interrupt();
        }
    }

    private void closeSockets() {
        connected = false;
        try {
            if (controlSocket != null && !controlSocket.isClosed()) {
                controlSocket.close();
            }
        } catch (Exception ignored) {}
        try {
            if (statusSocket != null && !statusSocket.isClosed()) {
                statusSocket.close();
            }
        } catch (Exception ignored) {}
    }

    private String bytesToHex(byte[] bytes) {
        StringBuilder sb = new StringBuilder();
        for (byte b : bytes) {
            sb.append(String.format("%02X ", b));
        }
        return sb.toString();
    }
}
