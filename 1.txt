public class ControlCardSocketHandler implements Runnable {
    // ... existing fields ...
    
    // Add health status command
    private final byte[] healthStatusCommand; // CO.HL.G command
    private final Consumer<HealthData> healthCallback; // New callback for health data
    
    // Update constructor to include health monitoring
    public ControlCardSocketHandler(String ip, int port, byte[] statusCommand, 
                                    Consumer<Boolean> statusCallback, 
                                    Consumer<HardwareData> dataCallback) {
        this.ip = ip;
        this.port = port;
        this.statusCommand = statusCommand;
        this.statusCallback = statusCallback;
        this.dataCallback = dataCallback;
        this.sendQueue = new ArrayBlockingQueue<>(100);
        
        // Default health command (CO.HL.G,1,0,0,50.00,30.00)
        this.healthStatusCommand = hexStringToBytes("43,4F,2E,48,4C,2E,47,2C,31,2C,30,2C,30,2C,35,30,2E,30,30,2C,33,30,2E,30,30");
        this.healthCallback = null;
    }
    
    // New constructor with health callback
    public ControlCardSocketHandler(String ip, int port, byte[] statusCommand, 
                                    Consumer<Boolean> statusCallback, 
                                    Consumer<HardwareData> dataCallback,
                                    byte[] healthCommand,
                                    Consumer<HealthData> healthCallback) {
        this.ip = ip;
        this.port = port;
        this.statusCommand = statusCommand;
        this.statusCallback = statusCallback;
        this.dataCallback = dataCallback;
        this.healthStatusCommand = healthCommand != null ? healthCommand : 
            hexStringToBytes("43,4F,2E,48,4C,2E,47,2C,31,2C,30,2C,30,2C,35,30,2E,30,30,2C,33,30,2E,30,30");
        this.healthCallback = healthCallback;
        this.sendQueue = new ArrayBlockingQueue<>(100);
    }
    
    // Update auto-polling to include health status
    private void startAutoPolling() {
        scheduler = Executors.newSingleThreadScheduledExecutor();
        
        // Poll for status command every 300ms (as specified)
        scheduler.scheduleAtFixedRate(() -> {
            if (connected.get() && statusCommand != null) {
                enqueueCommand(statusCommand);
            }
        }, 0, 300, TimeUnit.MILLISECONDS); // Changed to 300ms as per requirements
        
        // Poll for health status every 1 second
        scheduler.scheduleAtFixedRate(() -> {
            if (connected.get() && healthStatusCommand != null) {
                enqueueCommand(healthStatusCommand);
            }
        }, 0, 1, TimeUnit.SECONDS);
    }
    
    // Add method to parse health status response
    private HealthData parseHealthData(byte[] frame) {
        HealthData health = new HealthData();
        
        try {
            String response = new String(frame, "ASCII");
            
            // Parse response format: CO.HL.G,PAOnOff,VSWR,TEMP,FWD,RFL
            // Example: CO.HL.G,1,0,0,50.00,30.00
            if (response.startsWith("CO.HL.G")) {
                String[] parts = response.split(",");
                if (parts.length >= 6) {
                    health.paOn = parts[1].equals("1");
                    health.vswrAlarm = parts[2].equals("1");
                    health.tempAlarm = parts[3].equals("1");
                    health.fwdPower = Double.parseDouble(parts[4]);
                    health.refPower = Double.parseDouble(parts[5]);
                }
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
        
        return health;
    }
    
    // Update handleFrame to process health responses
    private void handleFrame(byte[] frame) {
        // Check if it's health status response
        String response = new String(frame, StandardCharsets.US_ASCII);
        if (response.contains("CO.HL.G")) {
            HealthData health = parseHealthData(frame);
            // Process health data
            processHealthData(health);
            return;
        }
        
        // Original frame processing for hardware data
        if (!verifyChecksum(frame)) {
            System.out.println("XOR failed" + bytesToHex(frame));
            return;
        }

        HardwareData data = parseFrame(frame);
        this.lastData = data;

        if (dataCallback != null) {
            dataCallback.accept(data);
        }

        System.out.println("VSWR: " + data.vswr + " TEMP: " + data.temp + 
                         " FWD: " + data.fwd + " RFL: " + data.rfl);
    }
    
    private void processHealthData(HealthData health) {
        if (healthCallback != null) {
            healthCallback.accept(health);
        }
        System.out.println("Health Status - PA: " + (health.paOn ? "ON" : "OFF") +
                         " | VSWR Alarm: " + health.vswrAlarm +
                         " | Temp Alarm: " + health.tempAlarm +
                         " | FWD: " + health.fwdPower + "W" +
                         " | RFL: " + health.refPower + "W");
    }
    
    // Helper method to convert hex string to bytes
    private byte[] hexStringToBytes(String hex) {
        if (hex == null || hex.trim().isEmpty()) {
            return new byte[0];
        }

        String[] parts = hex.trim().split(",");
        List<Byte> byteList = new ArrayList<>();

        for (String part : parts) {
            String value = part.trim();
            if (!value.isEmpty()) {
                byteList.add((byte) Integer.parseInt(value, 16));
            }
        }

        byte[] bytes = new byte[byteList.size()];
        for (int i = 0; i < byteList.size(); i++) {
            bytes[i] = byteList.get(i);
        }

        return bytes;
    }
}

public class HealthData {
    public boolean paOn;
    public boolean vswrAlarm;
    public boolean tempAlarm;
    public double fwdPower;
    public double refPower;
    public String timestamp;
    
    public HealthData() {
        this.paOn = false;
        this.vswrAlarm = false;
        this.tempAlarm = false;
        this.fwdPower = 0.0;
        this.refPower = 0.0;
        this.timestamp = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss"));
    }
    
    public double calculateVSWR() {
        if (fwdPower > 0 && refPower >= 0) {
            double gamma = Math.sqrt(refPower / fwdPower);
            if (gamma < 1.0) {
                return (1 + gamma) / (1 - gamma);
            }
        }
        return 1.0;
    }
    
    @Override
    public String toString() {
        return String.format("PA: %s | VSWR Alarm: %s | Temp Alarm: %s | FWD: %.2fW | RFL: %.2fW",
            paOn ? "ON" : "OFF", vswrAlarm ? "ACTIVE" : "OK", 
            tempAlarm ? "ACTIVE" : "OK", fwdPower, refPower);
    }
}

public class HealthMonitor {
    private final ScheduledExecutorService scheduler;
    private final Map<String, HealthData> healthDataMap;
    private final Map<String, ControlCardSocketHandler> handlers;
    
    public HealthMonitor() {
        this.scheduler = Executors.newSingleThreadScheduledExecutor();
        this.healthDataMap = new ConcurrentHashMap<>();
        this.handlers = new HashMap<>();
    }
    
    public void registerCard(String cardId, ControlCardSocketHandler handler) {
        handlers.put(cardId, handler);
        // Start health polling for this card
        scheduler.scheduleAtFixedRate(() -> {
            if (handler.isConnected()) {
                handler.enqueueCommand(hexStringToBytes("43,4F,2E,48,4C,2E,47,2C,31,2C,30,2C,30,2C,35,30,2E,30,30,2C,33,30,2E,30,30"));
            }
        }, 0, 1, TimeUnit.SECONDS);
    }
    
    public HealthData getHealthData(String cardId) {
        return healthDataMap.get(cardId);
    }
    
    public void updateHealthData(String cardId, HealthData data) {
        healthDataMap.put(cardId, data);
        // Log any alarms
        if (data.vswrAlarm || data.tempAlarm) {
            System.err.println("ALERT: " + cardId + " - " + data);
        }
    }
    
    public void shutdown() {
        scheduler.shutdown();
    }
}
